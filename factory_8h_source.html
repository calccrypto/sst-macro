152 class SpktFactory : public SpktFactoryTemplateBase<T>
153 {
154 
155  public:
156  typedef SpktFactoryTemplateBase<T> parent;
157  typedef T element_type;
158  using SpktFactoryTemplateBase<T>::object_map_;
159  using SpktFactoryTemplateBase<T>::alias_map_;
160  using SpktFactoryTemplateBase<T>::name_;
161 
162  static void
163  register_alias(const std::string& oldname, const std::string& newname){
164  if (!object_map_) {
165  object_map_ = new typename parent::descr_map;
166  }
167  if (!alias_map_){
168  alias_map_ = new typename parent::alias_map;
169  }
170 
171  SpktDesc_base* base = (*object_map_)[oldname];
172  if (!base){
173  (*alias_map_)[oldname].push_back(newname);
174  } else {
175  (*object_map_)[newname] = base;
176  }
177 
178  }
179 
180  protected:
181  static T*
182  _get_value(const std::string& valname) {
183  parent::validate_map(valname);
184 
185  std::map<std::string, SpktDesc_base*>::const_iterator it =
186  object_map_->find(valname), end = object_map_->end();
187 
188  if (it == end) {
189  std::cerr << "Valid factories are:" << std::endl;
190  for (it = object_map_->begin(); it != end; ++it) {
191  std::cerr << it->first << std::endl;
192  }
193  spkt_throw_printf(value_error, "could not find name %s for factory %s",
194  valname.c_str(), name_);
195  }
196 
197  SpktDesc_base* descr = it->second;
198  if (!descr) {
199  spkt_throw_printf(value_error,
200  "initialized name %s with null descr for factory %s",
201  valname.c_str(), name_);
202  }
203 
204  factory_type* p = descr->build();
205 #if SPKT_ENABLE_PARAM_RECORDING
206  p->recorder()->record_factory_init(parent::name_, valname.c_str());
207 #endif
208  T* cast = dynamic_cast<T*>(p);
209  return cast;
210  }
211 
212  static T*
213  _get_param(const std::string& param_name,
214  sim_parameters* params) {
215  T* rv = _get_value(parent::value(param_name, params));
216  return rv;
217 
218  }
219

221  _get_optional_param(const std::string& param_name,
222  const std::string& defval, sim_parameters* params) {

224  return rv;
225  }

233  }

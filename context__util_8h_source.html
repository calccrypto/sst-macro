33 /// Pack a void* argument into a pair of integers.
34 union voidptr {
35  intpair vpair;
36  void *vpointer;
37  voidptr() {
38  vpair.a = 0;
39  vpair.b = 0;
40  vpointer = 0;
41  }
42  voidptr(void *ptr) {
43  assert(sizeof(void*) <= (2*sizeof(int)));
44  vpair.a = 0;
45  vpair.b = 0;
46  vpointer = ptr;
47  }
48  voidptr(int a, int b) {
49  vpointer = 0;
50  vpair.a = a;
51  vpair.b = b;
